Appendix A: AWS CLI Commands Reference

Reference Updated: Dec 23, 2025

This appendix provides a comprehensive reference of AWS CLI commands used throughout the ZL application deployment process.

📋 Command Categories

  • EKS cluster management commands
  • IAM and security commands
  • EC2 and networking commands
  • ECR and container commands
  • RDS database commands
  • CloudFormation stack commands

🔐 IAM & Security Commands

Configure AWS CLI

# Configure AWS CLI with access keys
aws configure

# Set default region
aws configure set default.region us-east-1

# Verify credentials
aws sts get-caller-identity

Create IAM User

# Create IAM user for EKS operations
aws iam create-user --user-name zl-eks-user

# Create access key for the user
aws iam create-access-key --user-name zl-eks-user

# Attach managed policies
aws iam attach-user-policy --user-name zl-eks-user --policy-arn arn:aws:iam::aws:policy/AmazonEKSClusterPolicy
aws iam attach-user-policy --user-name zl-eks-user --policy-arn arn:aws:iam::aws:policy/AmazonEKSWorkerNodePolicy

Create IAM Role for Service Accounts (IRSA)

# Create OIDC provider for EKS cluster
aws eks describe-cluster --name zl-eks-cluster --query "cluster.identity.oidc.issuer" --output text
aws iam create-open-id-connect-provider --url [OIDC_URL] --thumbprint-list [THUMBPRINT] --client-id-list sts.amazonaws.com

# Create IAM role for ALB controller
aws iam create-role --role-name zl-alb-controller-role --assume-role-policy-document file://alb-trust-policy.json
aws iam attach-role-policy --role-name zl-alb-controller-role --policy-arn arn:aws:iam::aws:policy/AmazonEKSLoadBalancerControllerPolicy

☸️ EKS Cluster Commands

Create EKS Cluster

# Create EKS cluster using eksctl
eksctl create cluster --name zl-eks-cluster --region us-east-1 --nodegroup-name zl-nodes --node-type t3.medium --nodes 3 --nodes-min 1 --nodes-max 5

# Update kubeconfig
aws eks update-kubeconfig --name zl-eks-cluster --region us-east-1

# Verify cluster
kubectl get nodes
kubectl get pods -n kube-system

Cluster Information

# Get cluster details
aws eks describe-cluster --name zl-eks-cluster --region us-east-1

# List all clusters
aws eks list-clusters --region us-east-1

# Get cluster endpoint
aws eks describe-cluster --name zl-eks-cluster --query "cluster.endpoint" --output text

Node Group Management

# List node groups
aws eks list-nodegroups --cluster-name zl-eks-cluster --region us-east-1

# Describe node group
aws eks describe-nodegroup --cluster-name zl-eks-cluster --nodegroup-name zl-nodes --region us-east-1

# Update node group
aws eks update-nodegroup-config --cluster-name zl-eks-cluster --nodegroup-name zl-nodes --scaling-config minSize=2,maxSize=10,desiredSize=3 --region us-east-1

🐳 ECR Commands

Create ECR Repository

# Create ECR repository
aws ecr create-repository --repository-name ue1-zlps-ecr-01 --region us-east-1

# List repositories
aws ecr describe-repositories --region us-east-1

# Get repository URI
aws ecr describe-repositories --repository-names ue1-zlps-ecr-01 --query "repositories[0].repositoryUri" --output text

ECR Authentication & Push

# Login to ECR
aws ecr get-login-password --region us-east-1 | docker login --username AWS --password-stdin [ACCOUNT_ID].dkr.ecr.us-east-1.amazonaws.com

# Build and tag image
docker build -t zl-server:latest .
docker tag zl-server:latest [ACCOUNT_ID].dkr.ecr.us-east-1.amazonaws.com/ue1-zlps-ecr-01:zlserver-1.0.0

# Push image
docker push [ACCOUNT_ID].dkr.ecr.us-east-1.amazonaws.com/ue1-zlps-ecr-01:zlserver-1.0.0

List ECR Images

# List images in repository
aws ecr list-images --repository-name ue1-zlps-ecr-01 --region us-east-1

# Get image details
aws ecr describe-images --repository-name ue1-zlps-ecr-01 --image-ids imageTag=zlserver-1.0.0 --region us-east-1

# Delete image
aws ecr batch-delete-image --repository-name ue1-zlps-ecr-01 --image-ids imageTag=zlserver-1.0.0 --region us-east-1

🗄️ RDS Commands

Create RDS PostgreSQL

# Create RDS instance
aws rds create-db-instance --db-instance-identifier zl-postgres-db --db-instance-class db.t3.micro --engine postgres --master-username zladmin --master-user-password [PASSWORD] --allocated-storage 20 --vpc-security-group-ids sg-12345678 --db-subnet-group-name zl-db-subnet-group --region us-east-1

# Get RDS endpoint
aws rds describe-db-instances --db-instance-identifier zl-postgres-db --query "DBInstances[0].Endpoint.Address" --output text

RDS Management

# List DB instances
aws rds describe-db-instances --region us-east-1

# Modify DB instance
aws rds modify-db-instance --db-instance-identifier zl-postgres-db --apply-immediately --backup-retention-period 7 --region us-east-1

# Create DB snapshot
aws rds create-db-snapshot --db-instance-identifier zl-postgres-db --db-snapshot-identifier zl-postgres-snapshot --region us-east-1

🌐 VPC & Networking

VPC Information

# List VPCs
aws ec2 describe-vpcs --region us-east-1

# Get VPC details
aws ec2 describe-vpcs --vpc-ids vpc-12345678 --region us-east-1

# List subnets
aws ec2 describe-subnets --filters "Name=vpc-id,Values=vpc-12345678" --region us-east-1

Security Groups

# List security groups
aws ec2 describe-security-groups --region us-east-1

# Create security group
aws ec2 create-security-group --group-name zl-eks-sg --description "Security group for ZL EKS cluster" --vpc-id vpc-12345678 --region us-east-1

# Add inbound rule
aws ec2 authorize-security-group-ingress --group-id sg-12345678 --protocol tcp --port 443 --cidr 0.0.0.0/0 --region us-east-1

☁️ CloudFormation

Stack Operations

# List stacks
aws cloudformation list-stacks --region us-east-1

# Describe stack
aws cloudformation describe-stacks --stack-name zl-eks-stack --region us-east-1

# Get stack outputs
aws cloudformation describe-stacks --stack-name zl-eks-stack --query "Stacks[0].Outputs" --region us-east-1

# Delete stack
aws cloudformation delete-stack --stack-name zl-eks-stack --region us-east-1

🔍 Monitoring & Inspector

AWS Inspector

# Enable Inspector for ECR
aws inspector2 enable --resource-types ECR --region us-east-1

# List coverage (scan status)
aws inspector2 list-coverage --region us-east-1 --filter-criteria '{"resourceType":[{"comparison":"EQUALS","value":"AWS_ECR_CONTAINER_IMAGE"}]}'

# Get findings
aws inspector2 list-findings --region us-east-1 --filter-criteria '{"resourceType":[{"comparison":"EQUALS","value":"AWS_ECR_CONTAINER_IMAGE"}],"ecrImageRepositoryName":[{"comparison":"EQUALS","value":"ue1-zlps-ecr-01"}]}'
