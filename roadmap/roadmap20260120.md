### What's Working

1. **Draft Testing Flow** (`/api/deployment-drafts/:id/test`):
   - Validates cloud credentials against provider API
   - Checks configuration completeness
   - Stores test results in draft
   - Does NOT create any infrastructure

2. **Terraform Plan Preview** (`planTerraform()`):
   - Runs `terraform init` → `validate` → `plan`
   - Shows what WOULD be created/changed
   - Saves plan output to `previewResults` field
   - Supports `skipStatusChange` to avoid deployment record changes

3. **Cleanup Scripts**:
   - Support `--dry-run` flag to show what would be deleted

### Gap: No Unified "Dry Run Mode" Toggle

- ❌ A single "Dry Run Mode" checkbox in the UI

**localDeploymentService.js Structure:**

```javascript
class LocalDeploymentService {
  constructor() {
    this.manifestDir = path.join(process.env.TERRAFORM_WORKING_DIR || '/tmp/zlaws_deployments', 'local');
  }

  // Check if Minikube is running and kubectl context is correct
  async checkPrerequisites() {
    const context = await this.execCommand('kubectl config current-context');
    if (context.trim() !== 'minikube') {
      throw new Error(`kubectl context is '${context.trim()}', expected 'minikube'. Run: kubectl config use-context minikube`);
    }
    
    const status = await this.execCommand('minikube status --format={{.Host}}');
    if (status.trim() !== 'Running') {
      throw new Error('Minikube is not running. Start it with: minikube start');
    }
    
    return { context: 'minikube', status: 'running' };
  }

  // Generate K8s manifests from deployment configuration
  async generateManifests(config, deploymentId) {
    const generator = new KubernetesManifestGenerator(config);
    
    const manifests = {
      namespace: generator.generateNamespace(),
      configMaps: generator.generateConfigMaps(),
      secrets: generator.generateSecrets(),
      pvcs: generator.generatePVCs({ storageClass: 'standard' }), // Minikube default
      deployments: generator.generateDeployments(),
      services: generator.generateServices(),
      ingress: generator.generateIngress(),
    };
    
    // Write manifests to files
    const manifestPath = path.join(this.manifestDir, deploymentId);
    await this.writeManifests(manifestPath, manifests);
    
    return { path: manifestPath, manifests };
  }

  // Apply manifests to Minikube
  async applyManifests(manifestPath, deploymentId, emitProgress) {
    const order = ['namespace.yaml', 'configmaps/', 'secrets/', 'pvcs/', 'deployments/', 'services/', 'ingress.yaml'];
    
    for (let i = 0; i < order.length; i++) {
      const file = path.join(manifestPath, order[i]);
      if (fs.existsSync(file)) {
        await this.execCommand(`kubectl apply -f ${file}`);
        emitProgress(Math.round(((i + 1) / order.length) * 80) + 10);
      }
    }
  }

  // Wait for deployment to be ready
  async waitForReady(namespace, deploymentId, timeout = 300000) {
    const start = Date.now();
    while (Date.now() - start < timeout) {
      const result = await this.execCommand(
        `kubectl get pods -n ${namespace} -o jsonpath='{.items[*].status.phase}'`
      );
      const phases = result.trim().split(' ');
      if (phases.every(p => p === 'Running')) {
        return true;
      }
      await new Promise(r => setTimeout(r, 5000));
    }
    throw new Error('Timeout waiting for pods to be ready');
  }

  // Get access URLs
  async getAccessUrls(namespace, serviceName) {
    const url = await this.execCommand(`minikube service ${serviceName} -n ${namespace} --url`);
    return { serviceUrl: url.trim() };
  }
}
```

**kubernetesManifestGenerator.js Structure:**

```javascript
class KubernetesManifestGenerator {
  constructor(config) {
    this.config = config;
    this.namespace = config.clusterName || 'zlaws-local';
  }

  generateNamespace() {
    return {
      apiVersion: 'v1',
      kind: 'Namespace',
      metadata: { name: this.namespace }
    };
  }

  generateDeployments() {
    const deployments = [];
    
    // Backend deployment
    deployments.push({
      apiVersion: 'apps/v1',
      kind: 'Deployment',
      metadata: {
        name: 'zlaws-backend',
        namespace: this.namespace,
      },
      spec: {
        replicas: this.config.nodeCount || 1,
        selector: { matchLabels: { app: 'zlaws-backend' } },
        template: {
          metadata: { labels: { app: 'zlaws-backend' } },
          spec: {
            containers: [{
              name: 'backend',
              image: 'zlaws:latest',
              imagePullPolicy: 'Never', // Use local image
              ports: [{ containerPort: 5000 }],
              envFrom: [{ configMapRef: { name: 'zlaws-config' } }],
            }]
          }
        }
      }
    });
    
    // Database deployment (PostgreSQL)
    if (this.config.database?.enabled) {
      deployments.push(this.generatePostgresDeployment());
    }
    
    return deployments;
  }

  generateServices() {
    return [{
      apiVersion: 'v1',
      kind: 'Service',
      metadata: {
        name: 'zlaws-backend',
        namespace: this.namespace,
      },
      spec: {
        type: 'NodePort',
        ports: [{ port: 80, targetPort: 5000 }],
        selector: { app: 'zlaws-backend' }
      }
    }];
  }

  // ... additional generators for ConfigMaps, Secrets, PVCs, etc.
}
```

### Task 4: Add Local Environment Check Endpoint

**File:** `backend/src/routes/local-deployments.js`

```javascript
const express = require('express');
const router = express.Router();
const { exec } = require('child_process');
const { promisify } = require('util');
const execAsync = promisify(exec);

// GET /api/local-deployments/status
// Check local environment readiness
router.get('/status', async (req, res) => {
  try {
    const status = {
      minikube: { status: 'unknown', version: null },
      kubectl: { context: null, isMinikube: false },
      docker: { running: false },
    };

    // Check Minikube
    try {
      const { stdout: mkStatus } = await execAsync('minikube status --format={{.Host}}');
      status.minikube.status = mkStatus.trim() === 'Running' ? 'running' : 'stopped';
      
      const { stdout: mkVersion } = await execAsync('minikube version --short');
      status.minikube.version = mkVersion.trim();
    } catch (e) {
      status.minikube.status = 'not-installed';
    }

    // Check kubectl context
    try {
      const { stdout: context } = await execAsync('kubectl config current-context');
      status.kubectl.context = context.trim();
      status.kubectl.isMinikube = context.trim() === 'minikube';
    } catch (e) {
      status.kubectl.context = null;
    }

    // Check Docker
    try {
      await execAsync('docker info');
      status.docker.running = true;
    } catch (e) {
      status.docker.running = false;
    }

    const isReady = status.minikube.status === 'running' && 
                    status.kubectl.isMinikube && 
                    status.docker.running;

    res.json({ 
      success: true, 
      isReady,
      ...status,
      message: isReady ? 'Local environment ready' : 'Local environment not ready'
    });
  } catch (error) {
    res.status(500).json({ success: false, error: error.message });
  }
});

// POST /api/local-deployments/start-minikube
// Start Minikube if not running
router.post('/start-minikube', async (req, res) => {
  try {
    await execAsync('minikube start --cpus=4 --memory=4096');
    res.json({ success: true, message: 'Minikube started' });
  } catch (error) {
    res.status(500).json({ success: false, error: error.message });
  }
});

// POST /api/local-deployments/switch-context
// Switch kubectl context to minikube
router.post('/switch-context', async (req, res) => {
  try {
    await execAsync('kubectl config use-context minikube');
    res.json({ success: true, message: 'Switched to minikube context' });
  } catch (error) {
    res.status(500).json({ success: false, error: error.message });
  }
});

module.exports = router;
```
